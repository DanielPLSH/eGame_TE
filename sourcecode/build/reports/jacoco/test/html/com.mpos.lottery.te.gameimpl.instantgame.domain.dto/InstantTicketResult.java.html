package com.mpos.lottery.te.gameimpl.instantgame.domain.dto;

import java.math.BigDecimal;

public class InstantTicketResult {

        this.totalAmount = totalAmount;
    }

    public int getCode() {
        return code;
    }

    public void setCode(int code) {
        this.code = code;
    }

}
